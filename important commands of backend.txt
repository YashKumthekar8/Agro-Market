3) Select plants/fertilizers based on location
	db.selling_app_plant.find({'location_id': __ })
	
4) Select location based on plant/fertilizers
	id = db.selling_app_plant.find({'id': __ }, {location_id:1, _id:0})
	db.selling_app_location.find({'id', id })
